28. Reverse a Number (3 Ways)

public class ReverseNumber {
    public static void main(String[] args) {
        int num = 1234;
        
        // Method 1: Using a loop
        int reversed = 0;
        while (num != 0) {
            reversed = reversed * 10 + num % 10;
            num /= 10;
        }
        System.out.println("Method 1: Reversed Number: " + reversed);
        
        // Method 2: Using StringBuilder
        String str = Integer.toString(num);
        reversed = Integer.parseInt(new StringBuilder(str).reverse().toString());
        System.out.println("Method 2: Reversed Number: " + reversed);
        
        // Method 3: Using recursion
        reversed = reverseRecursive(num, 0);
        System.out.println("Method 3: Reversed Number: " + reversed);
    }
    
    public static int reverseRecursive(int num, int reversed) {
        if (num == 0) {
            return reversed;
        }
        return reverseRecursive(num / 10, reversed * 10 + num % 10);
    }
}

Output:

Method 1: Reversed Number: 4321
Method 2: Reversed Number: 4321
Method 3: Reversed Number: 4321
29. Reverse a String (3 Ways)

public class ReverseString {
    public static void main(String[] args) {
        String str = "Hello";
        
        // Method 1: Using StringBuilder
        String reversed = new StringBuilder(str).reverse().toString();
        System.out.println("Method 1: Reversed String: " + reversed);
        
        // Method 2: Using a loop
        reversed = "";
        for (int i = str.length() - 1; i >= 0; i--) {
            reversed += str.charAt(i);
        }
        System.out.println("Method 2: Reversed String: " + reversed);
        
        // Method 3: Using recursion
        reversed = reverseRecursive(str);
        System.out.println("Method 3: Reversed String: " + reversed);
    }
    
    public static String reverseRecursive(String str) {
        if (str.isEmpty()) {
            return str;
        }
        return reverseRecursive(str.substring(1)) + str.charAt(0);
    }
}

Output:

Method 1: Reversed String: olleH
Method 2: Reversed String: olleH
Method 3: Reversed String: olleH

import java.io.FileReader;
import java.io.IOException;

public class ReadFile {
    public static void main(String[] args) {
        try (BufferedReader br = new BufferedReader(new FileReader("input.txt"))) {
            String line;
